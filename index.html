輸入框readonly及清除鈕

view source 
<!doctype html>
<html>
<head>
    <link href="css/base/jquery-ui.min.css" rel="stylesheet">
    <style>
        button.ui-datepicker-current {
            display: none;
        }
    </style>
</head>
<body>	
<input type="text" class="datepicker">	
</body>
<script src="https://code.jquery.com/jquery-1.11.3.js"></script>
<script src="js/jquery-ui.min.js"></script>
<script src="js/jquery.ui.datepicker-zh-TW.js"></script>
<script>
    $(function () {
        var originalPicker = $.datepicker._generateHTML;
        $.datepicker._generateHTML = function (inst) {
            var $picker = $(originalPicker.call($.datepicker, inst)); //call the original
            $picker = $('<div>').append($picker); //add a wrapper div for jQuery context
            $('.ui-datepicker-buttonpane', $picker).append(
                $('<button class="ui-datepicker-clear ui-state-default ui-priority-primary ui-corner-all">' + (this._get(inst,
                    'clear') === undefined ? 'Clear' : this._get(inst, 'clear')) + '</button>')
                .click(function () {
                    inst.input.val('').datepicker('hide');
                })
            );
 
            return $picker.children();
        };
 
        $('.datepicker').datepicker({
            showButtonPanel: true
        }).find(".ui-datepicker-current").hide();
    });
</script>
</html>
